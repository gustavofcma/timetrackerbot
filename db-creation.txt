CREATE TABLE ENTRIES (
  DAY_ID  NUMERIC NOT NULL DEFAULT CURRENT_DATE UNIQUE,
  USER_ID  TEXT NOT NULL,
  ENTRANCE  NUMERIC,
  LUNCH  NUMERIC,
  LUNCHRETURN  NUMERIC,
  LEAVE  NUMERIC,
  PRIMARY KEY(DAY_ID,USER_ID)
);

CREATE TABLE USERS (
  USER_ID  TEXT NOT NULL UNIQUE,
  USER_NAME  TEXT,
  USER_HANDLE  TEXT,
  PRIMARY KEY(USER_ID)
);
